make string to like backquote

ES6 字符串 转 模板字符串. 如何得出一个类似 模版字符串变量 的效果。 

const util = require('util')

function tmplParse(tmp,paramObj){
  var rtn = ""
  var params = []
  var tmplPos = 0
  var beginPos,endPos = 0
  var key,value
  while((beginPos = tmp.indexOf('${',endPos)) > -1){
    endPos = tmp.indexOf('}',beginPos)

    if (endPos > -1){
      key = tmp.substring(beginPos+2,endPos)
      value = paramObj[key]?paramObj[key]:''
      rtn += tmp.substring(tmplPos,beginPos)
      rtn += typeof value === 'number'?"%d":"%s"
      params.push(value)
      endPos++
      tmplPos = endPos
    }
    else
      break
  }
  if (endPos != -1)
    rtn += tmp.substr(endPos)
  console.log(rtn)
  console.log(params)
  return util.format(rtn,...params)
}

var obj = {name:'cdn',age:100}

var tmplName = "my name is ${name},age is ${age}, next name ${name}"
var str = tmplParse(tmplName,obj)
console.log(str)
