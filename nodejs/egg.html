egg with postgresql

sequelize queries returning only strings

当数据库是 postgresql 时，为什么，取到的数据类型是 string, 而不是 int ？

  • https://github.com/sequelize/sequelize/issues/2383#issuecomment-58006083
  • https://github.com/brianc/node-postgres/pull/353

https://github.com/brianc/node-postgres/pull/353#issuecomment-283709264 

var pg = require('pg');

// bigint
pg.types.setTypeParser(20, function (value) {
    return parseInt(value);
});

// numeric
pg.types.setTypeParser(1700, function (value) {
    return parseFloat(value);
});

so, need add above code snippet into app.js. 

'use strict';
const start = Date.now();
module.exports = app => {
  app.logger.info('启动耗时 %d ms', Date.now() - start);
  app.beforeStart(async () => {
    const ctx = app.createAnonymousContext();
    // 初始化延迟任务
    app.initDelayTask();

    var pg = require('pg');
    // bigint
    pg.types.setTypeParser(20, function (value) {
        return parseInt(value);
    });
    // numeric
    pg.types.setTypeParser(1700, function (value) {
        return parseFloat(value);
    });
  });
  app.logger.info('process.env.NODE_OPTIONS is: ' + process.env.NODE_OPTIONS);
};

It's effective.
