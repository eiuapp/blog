mysql.app be true or false

如果把 config/config.default.js 下的 mysql.app 设置成 true, 则只要app启动时,就 会检查. 所以, 当你不再使用mysql时, 而又要在 config/config.default.js 中保留这个配置时, 则要把 mysql.app 设置成 false . 

// 数据库
mysql: {
  client: {
    host: '192.168.168.137',
    // 端口号
    port: '3306',
    // 用户名
    user: 'esop',
    // 密码
    password: 'CwfhbsgL0RE7upUx',
    // 数据库名
    database: 'esop',
  },
  // 所有数据库配置的默认值
  default: {},

  // 是否加载到 app 上，默认开启
  app: false,
  // 是否加载到 agent 上，默认关闭
  agent: false,
},
