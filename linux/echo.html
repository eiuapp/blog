输出不自动添加换行

-n 表示echo输出不自动添加换行

示例: 

#!/bin/bash
seedJsDir="../../seeders/";
sqlFileDir="../sql/";
for file in `ls ${sqlFileDir}`; #  | head -2
do
    echo ${file}
    table=${file%.*}
    echo ${table}
    cur_dateTime="`date +%Y%m%d%H%m%S`"
    seedJsFile=${seedJsDir}${cur_dateTime}-add-data-${table}.js
    cat ./seed-head.txt > ${seedJsFile}
    cat ${sqlFileDir}${file} >> ${seedJsFile}
    echo -n `cat ./seed-middle.txt` >> ${seedJsFile}
    cat ./seed-p3.txt >> ${seedJsFile}
    echo -n  ${table}  >> ${seedJsFile}
    cat ./seed-tail.txt >> ${seedJsFile}
done
