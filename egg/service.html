egg service throw error

先在 app/service/a.js 中写下 ctx.status, ctx.body,

if ('' + id === '' + userInfo.id) {
  // return this.fail(41001, '自己不可以删自己');
  ctx.status = 200;
  ctx.body = {
    code: 41001,
    message: '自己不可以删自己！',
  };
  ctx.throw(41001, ctx.body.message);
}

在 app/controller/a.js 中写下 catch 接收 err

try {
  const usm_user_id = await ctx.service.umsUser.deleteUserById({ id: idArray[0] });
  return this.success(usm_user_id);
} catch (err) {
  this.ctx.logger.info(`err is ${JSON.stringify(err)}, ctx.body.code: ${this.ctx.body.code}`);
  return this.fail(this.ctx.body.code, this.ctx.body.message);
}

(可选) 在 app/core/base_controller.js 中写下 fail, success

success(data, status) {
  this.ctx.body = { code: this.ctx.SUCCESS_CODE, data };
  this.ctx.status = status || 200;
}

fail(code, message) {
  this.ctx.body = { code, message, data: {} };
  this.ctx.status = 200;
}

notFound(msg) {
  msg = msg || 'not found';
  this.ctx.throw(404, msg);
}
