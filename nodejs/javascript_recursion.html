js 递归遍历节点下的所有children,直到节点下不存在children

https://www.cnblogs.com/qdkfyym/p/13254448.html 

function calleArr(array) {
    for(i in array){
        var data=array[i];
        if (data.children) {
            data.name=data.name+"--111";
            calleArr(data.children) //自己调用自己
        } else {
            data.name=data.name+"--111";
        }
    }
}
