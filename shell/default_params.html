for + default parameters

如果不加参数则默认执行restart函数，加stop，则执行stop函数

https://www.cnblogs.com/outsrkem/p/11196724.html

#+BEGIN_SRC bash -l 7 :hl_lines 8 #!/bin/bash

strategys=( "near_stop_out_dynamic_sleep" "retracement_touch_max_loss_amount_dynamic_sleep_symbol" "anti_trend_close_plus" )

function stop(){

echo "stop() start …"

for dir in ${strategys[@]};do echo "\(dir stop ..." ps -ef | grep "main" | grep "\){dir}" | awk '{print $2}' | xargs kill done

echo "stop() end …" } function start(){

echo "start() start …"

for dir in ${strategys[@]};do echo "\(dir for start ..." ps -ef | grep "main" | grep "\){dir}" | awk '{print $2}' | xargs kill go build -o ${dir}/main.build \({dir}/main.go && nohup ./\){dir}/main.build > ${dir}/`date +%Y%m%d%H%M%S`.log 2>&1 &

echo "for next …" sleep 1 done

echo "start() end …" } function restart(){

echo "restart() start …"

stop $strategys sleep 3 start $strategys

echo "restart() end …" }

cmd=$1

echo "start ……………." ${cmd:-restart}

#END_SRC
