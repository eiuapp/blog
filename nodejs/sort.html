array sort

'use strict';
const actions = [
  {
    action: 'list',
    action_id: 5,
    action_name: '查询-账号列表',
  },
  {
    action: 'create',
    action_id: 6,
    action_name: '增加-账号信息',
  },
  {
    action: 'delete',
    action_id: 7,
    action_name: '删除-账号信息',
  },
  {
    action: 'update',
    action_id: 8,
    action_name: '更新-账号信息',
  },
  {
    action: 'id',
    action_id: 9,
    action_name: '查询-账号详情',
  },
  {
    action: 'passInit',
    action_id: 300,
    action_name: '初始化-账号密码',
  },
];
console.log(`actions: ${JSON.stringify(actions)}`);
let arr = actions;
const rule = { list: 1, create: 2, update: 3, id: 4, delete: 5 };
// 把 actions 按 actions[i].action 的值进行排序
arr = arr.sort((n1, n2) => {
  // return -1; //返回负值  交换顺序
  // return 0 或者 1   //返回正值   保持顺序不变
  console.log(n1, n2);
  return rule[n1.action] - rule[n2.action];
  // n1 - n2  从小到大
  // n2 - n1  从大到小
});
console.log(`arr: ${JSON.stringify(arr)}`);
console.log(`actions: ${JSON.stringify(actions)}`);
